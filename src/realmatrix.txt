fn main() {
    // A simple 3x3 matrix
    let matrix: [[char; 3]; 3] = [
        ['a', 'b', 'c'],
        ['d', 'e', 'f'],
        ['g', 'h', 'i']
    ];
    
    // Print it
    for row in matrix.iter() {
        for &ch in row.iter() {
            print!("{} ", ch);
        }
        println!();
    }
}
